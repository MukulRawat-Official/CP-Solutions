void solve()
{
  int x1, x2, y1, y2;

  cin >> x1 >> x2 >> y1 >> y2;

  if (x2 > y2)
  {
    swap(y2, x2);
    swap(x1, y1);
  }

  if (x2 < y1)
    cout << x1 + y1 << endl;

  else if (x1 >= y1)
    cout << x1 << endl;

  else
    cout << y1 << endl;

}
