void solve()
{
  int n;
  cin >> n;
  cin.ignore();
  string s;
  getline(cin, s);


  int c = n / 2;

  int left = c - 1;
  int right = c + 1;

  int count = 1;

  while (left >= 0)
  {
    if (s[left] == s[c])
      count++;

    else break;

    left--;
  }

  while (right < n)
  {
    if (s[right] == s[c])
      count++;

    else break;

    right++;
  }

  cout << count << endl;

}



int main()
{
  predefined();

  long t;

  cin >> t;

  while (t--)
    solve();
}
