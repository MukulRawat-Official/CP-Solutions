/* arr in reverse and from front ordere will be same as factor count from (1 , 2 ) or (2 , 1) so well search in only from front order
  just compare previus and current diff  if they are in ration 1:2 or 2:1 we're good to go then.

*/
void solve()
{
  int n;

  cin >> n;

  vector<long>arr(n);

  for (int i = 0; i < n; i++)
    cin >> arr[i];

  sort(arr.begin(), arr.end());

  ll curr = arr[1] - arr[0];
  debug(curr);

  bool p = true;
  for (int i = 2; i < n; i++)
  {
    long x = arr[i] - arr[i - 1];
    if (x != 2 * curr && 2 * x != curr)
    {
      p  = !p;
      break;
    }
    curr = x;
  }
  debug(p);



  if (p)
    cout << "yes" << endl;

  else
    cout << "no" << endl;
}


int main()
{
  predefined();
  lli t;

  cin >> t;

  while (t--)
    solve();

}
