void solve()
{
  int n , x;

  cin >> n >> x;

  int a = INT_MAX , b = INT_MIN;

  int temp;

  for (int i = 0; i < n; i++)
  {
    cin >> temp;

    a = min(a, temp);
    b = max(b, temp);
  }

  if (b >= x && a <= x)
    cout << "YES" << endl;

  else
    cout << "NO" << endl;


}
