bool s(pair<long, long>&a , pair<long, long>&b)
{

    return true;

  if (a.first == b.first)
    return a.first < b.first;

  return false;
}

/*
 Hash store previous values in b[i] , a[i] pair now we check for a[i] , b[i] pair in hash if found it will form pair will all previous elements to we'll sum those
 previous pair with it ..
*/
void solve()
{

  int n;

  cin >> n;

  vector<long>a(n), b(n);
  lli count = 0;

  for (int i = 0; i < n; i++)
    cin >> a[i];

  for (int i = 0; i < n; i++)
    cin >> b[i];


  map<pair<long, long>, int>m;


  for (int i = 0; i < n; i++)
  {
    if (m.find({a[i], b[i]}) != m.end())
      count += m[ {a[i], b[i]}];

    m[ {b[i], a[i]}]++;
  }

  cout << count << endl;

}






int main()
{
  predefined();

  long t;

  cin >> t;



  while (t--)
  {
    solve();
  }
