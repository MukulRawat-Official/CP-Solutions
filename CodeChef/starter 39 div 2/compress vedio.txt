/*
just count when factor changes from one element to another 

*/


void solve()
{
  int n , temp;

  unordered_set<int>s;

  cin >> n;

  cin >> temp;

  if (n == 1)
  {
    cout << 1 << endl;
    return;
  }


  n--;
  s.insert(temp);
  int count = 1;

  while (n--)
  {
    cin >> temp;

    if (s.find(temp) == s.end())
    {
      count++;
      s.clear();
      s.insert(temp);
    }
  }

  cout << count << endl;

}

int main()
{
  predefined();


  lli t;

  cin >> t;

  while (t--)
    solve();
}
