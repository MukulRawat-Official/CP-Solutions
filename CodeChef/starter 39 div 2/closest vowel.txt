void solve()
{
  int n;
  string s;

  vector<char>arr = {'a', 'e', 'i', 'o', 'u'};
  cin >> n;
  cin.ignore();
  getline(cin, s);

  vector<ll>dp(n + 1 , 0);

  dp[0] = 1;

  for (int i = 0; i < n; i++)
  {
    int min = INT_MAX;
    char ch = s[i];

    for (int j = 0; j < 5; j++)
    {
      if (min > abs(arr[j] - ch))
      {
        dp[i + 1] = 1;
        min = abs(arr[j] - ch);
      }

      else if (min == abs(arr[j] - ch))
        dp[i + 1]++;
    }


    dp[i + 1] *= dp[i];
    dp[i + 1] = dp[i + 1] % (1000000007);
  }

  cout << dp[n] << endl;

}

int main()
{
  predefined();


  lli t;

  cin >> t;

  while (t--)
    solve();
}
