// (a^b) & c === (a&c) ^ (a&b)
// a ^ a = 0
//so find all pairs such that  a & x and store in that with its frequency as they all will form answer
// then display result
void solve()
{
  int n;
  cin >> n;

  ll x;


  vector<ll>arr(n, 0);

  for (int i = 0; i < n; i++)
    cin >> arr[i];
  cin >> x;
  map<ll, ll>m;

  for (auto i : arr)
  {
    m[i & x]++;
  }


  ll ans = 0;
  for (auto i : m)
  {
    ans += i.second * i.second;
  }

  cout << ans << endl;
}
int main()
{
  predefined();


  lli t;

  cin >> t;

  while (t--)
  {
    solve();
  }

}
