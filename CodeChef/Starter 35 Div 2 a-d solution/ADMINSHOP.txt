/*
  if there are n shops then min n hours will be needed as admin visit one by one
  now if shop has size m then time required to handle all admin in that shop will be 
   == (x / m) + (x % m != 0)? 1 : 0 ;  // eg  x = 6   m = 2 so time  = 3 hr or x = 5 m = 2 so time again is 3 hr
   
   ans will be maximum of both of this 
*/

void solve()
{
  ll n , x;

  cin >> n >> x;

  ll temp = 0;
  ll maxm = n;
  for (int i = 0; i < n; i++)
  {
    cin >> temp;
    maxm =  max(maxm, (x / temp) + (x % temp != 0 ? 1 : 0));
  }

  cout << maxm << endl;

}
int main()
{
  predefined();


  lli t;

  cin >> t;

  while (t--)
  {
    solve();
  }

}
